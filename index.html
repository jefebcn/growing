      🔒 Firebase Storage Security Rules
      ▼
    

    

      


// storage.rules — GrowManager v3
// Ogni utente può leggere/scrivere SOLO nella propria
// cartella: users/{userId}/photos/*
// Limite: 5 MB · solo immagini

rules_version = '2';

service firebase.storage {
  match /b/{bucket}/o {

    // ── Blocca tutto di default ──────────────────
    match /{allPaths=**} {
      allow read, write: if false;
    }

    // ── Cartella personale utente ─────────────────
    // Path: users/{userId}/photos/{photoId}
    match /users/{userId}/photos/{photoId} {

      // Lettura: solo il proprietario
      allow read: if request.auth != null
                  && request.auth.uid == userId;

      // Upload: autenticato + stessa cartella + ≤5MB + solo immagini
      allow create: if request.auth != null
                  && request.auth.uid == userId
                  && request.resource.size <= 5 * 1024 * 1024
                  && request.resource.contentType
                       .matches('image/.*');

      // Eliminazione: solo il proprietario
      allow delete: if request.auth != null
                  && request.auth.uid == userId;

      // Nessun update diretto (sostituire eliminando e ricaricando)
      allow update: if false;
    }
  }
}

      

      

        
💡

        

          Lato client la compressione riduce già il payload
          Ma le Security Rules restano il gate definitivo: anche un file già sotto 5 MB in uscita dal canvas viene controllato da Firebase prima di essere accettato.

      🔥 Query Firestore — come funziona
      ▼
    

    

      


// Struttura documento in Firestore:
// Collection: users/{uid}/plants/{plantId}/photos/{photoId}

// ── 1. Carica tutte le foto di una pianta ──────────────
const loadPlantPhotos = async (uid, plantId) => {
  const photosRef = collection(db,
    `users/${uid}/plants/${plantId}/photos`
  );
  const q = query(
    photosRef,
    orderBy('createdAt', 'desc'),
    limit(50)
  );
  const snap = await getDocs(q);
  return snap.docs.map(d => ({ id: d.id, ...d.data() }));
};

// ── 2. Ascolto realtime su tutte le piante ─────────────
// (usato per aggiornare la UI senza reload)
const subscribePhotos = (uid, plantId, onUpdate) => {
  const q = query(
    collection(db, `users/${uid}/plants/${plantId}/photos`),
    orderBy('createdAt', 'desc')
  );
  return onSnapshot(q, snap => {
    const photos = snap.docs.map(d => ({ id: d.id, ...d.data() }));
    onUpdate(photos);
  });
};

// ── 3. Salva metadati dopo upload su Storage ───────────
const savePhotoDoc = async (uid, plantId, data) => {
  const ref = doc(
    collection(db, `users/${uid}/plants/${plantId}/photos`)
  );
  await setDoc(ref, {
    storageUrl:   data.url,
    storagePath:  data.path,
    plantId,
    fase:         data.fase,
    caption:      data.caption,
    origSize:     data.origSize,
    compSize:     data.compSize,
    width:        data.width,
    height:       data.height,
    createdAt:    serverTimestamp(),
  });
  return ref.id;
};
